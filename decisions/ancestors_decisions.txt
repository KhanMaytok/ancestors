targetted_decisions = {
	# Select a new target for Vee. Target must be at Pentaghast society leader court
	select_new_vee_host = {
		ai_check_interval = 12
		filter = court
		ai_target_filter = court
		is_high_prio = yes

		from_potential = {
			is_society_grandmaster = yes
			society_member_of = pentaghast_society	
  		}

		potential = {
    		is_female = yes
    		is_incapable = no
			prisoner = no
    		OR = {
    			trait = sharingan
    		}
    		NOT = { trait = syrena }
  		}

		allow = {
			has_global_flag = vanished_vee
			FROM = { is_liege_of = ROOT }
		}

  		effect = {
  			clr_global_flag = vanished_vee
  			ROOT = {
  				set_character_flag = is_vee
				add_trait = syrena
  			}
  		}

  		ai_will_do = {
		    factor = 1
		}
	}

	get_mangekyo_sharingan = {
		ai_check_interval = 12
		filter = friends
		ai_target_filter = friends

		from_potential = {
			trait = sharingan
			NOT = { trait = mangekyo_sharingan }
			NOT = { has_character_flag = wants_to_get_mangekyo }
		}

		potential = {
			NOT = { has_character_flag = target_for_mangekyo }
			is_friend = FROM
		}

		effect = {
		    FROM = { set_character_flag = wants_to_get_mangekyo }
		    ROOT = {
		    	set_character_flag = target_for_mangekyo
		    	character_event = { id = ACS.400 days = 2 }
		    }
		}

		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { trait = ambitious }
			}			
		}
	}

	hypnotize_with_sharingan = {
		ai_target_filter = sub_realm
		ai_check_interval = 12

		from_potential = {
			OR = {
				trait = sharingan
				trait = mangekyo_sharingan
			}			
		}

		allow = {
			FROM = {
				prisoner = no
				NOT = { trait = incapable }
				NOT = { has_character_flag = use_sharingan_for_hypnotic }
			}
			ROOT = {
				NOT = { has_character_flag = hypnotized_with_sharingan }
			}
		}

		effect = {
			ROOT = {
				opinion = {
					modifier = opinion_hypnotized_with_sharingan
					who = FROM
					years = 3
				}				
				set_character_flag = hypnotized_with_sharingan
			}
			
			FROM = {
				set_character_flag = use_sharingan_for_hypnotic
				add_trait = stressed
				add_favor = ROOT
			}
		}
	}
}
